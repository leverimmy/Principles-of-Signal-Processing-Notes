\subsection{作业 11}

\begin{homework}
    用直接 I 型和直接 II 型（标准型）结构实现以下系统函数：
    \begin{align*}
        H(z) = \frac{3 + 4.2z^{-1}+0.8z^{-2}}{2 + 0.6z^{-1} - 0.4z^{-2}}.
    \end{align*}
\end{homework}

\begin{homework}
    已知某系统的差分方程如下式：
    \begin{align*}
        y(n) = 2x(n) + x(n - 3) - 0.9y(n - 1) + 0.36y(n - 2).
    \end{align*}
    \begin{enumerate}[label=(\arabic*)]
        \item 求该系统的传递函数 $H(z)$。
        \item 画出该系统的信号流图。
        \item 求该系统对应的因果序列及其收敛域，并判断该状态下系统是否稳定。
    \end{enumerate}
\end{homework}

\begin{homework}
    已知离散系统的差分方程为
    \begin{align*}
        y(n) = x(n) + 4x(n - 1) + 0.7y(n - 1) - 0.1y(n - 2),
    \end{align*}
    求：
    \begin{enumerate}[label=(\arabic*)]
        \item 该系统的传递函数 $H(z)$。
        \item 系统的单位冲激响应 $h(n)$。
        \item 画出系统的零极点分布。
        \item 说明系统的高低通特性。
        \item 说明系统的稳定性。
    \end{enumerate}
\end{homework}
